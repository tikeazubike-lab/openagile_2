### API Integration Approach

#### API Client Structure

Create a new API client module:

```typescript
// src/lib/api-client.ts
import { frappe } from 'frappe-js-sdk';

// Initialize Frappe client
const frappeClient = frappe({
  url: process.env.VITE_FRAPPE_URL || 'https://tuts.erpnext.zubbystudio.shop',
  token: () => localStorage.getItem('frappe_token'),
});

// API methods for each DocType
export const api = {
  // Tutors
  getTutors: (params?: any) => frappeClient.getDocList('Tutor', params),
  getTutor: (name: string) => frappeClient.getDoc('Tutor', name),
  createTutor: (data: any) => frappeClient.insertDoc('Tutor', data),
  
  // Students
  getStudents: (params?: any) => frappeClient.getDocList('Student', params),
  getStudent: (name: string) => frappeClient.getDoc('Student', name),
  
  // Sessions
  getSessions: (params?: any) => frappeClient.getDocList('Tutoring Session', params),
  createSession: (data: any) => frappeClient.insertDoc('Tutoring Session', data),
  
  // Bookings
  getBookings: (params?: any) => frappeClient.getDocList('Booking', params),
  createBooking: (data: any) => frappeClient.insertDoc('Booking', data),
  
  // Payments
  getPayments: (params?: any) => frappeClient.getDocList('Payment', params),
  createPayment: (data: any) => frappeClient.insertDoc('Payment', data),
  
  // Course Packages
  getPackages: (params?: any) => frappeClient.getDocList('Course Package', params),
  
  // Subscriptions
  getSubscriptions: (params?: any) => frappeClient.getDocList('Subscription', params),

  getGroupSessions: (params?: any) => frappeClient.getDocList('Group Session', params),
  
  // Custom API methods
  getLandingPageData: () => frappeClient.call('edu_theme.api.get_landing_page_data'),

```

#### TanStack Query Hooks

Create custom hooks for each data type:

```typescript
// src/hooks/use-tutors.ts
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { api } from '@/lib/api-client';

export const useTutors = (params?: any) => {
  return useQuery({
    queryKey: ['tutors', params],
    queryFn: () => api.getTutors(params),
  });
};

export const useTutor = (name: string) => {
  return useQuery({
    queryKey: ['tutor', name],
    queryFn: () => api.getTutor(name),
    enabled: !!name,
  });
};

export const useCreateTutor = () => {
  const queryClient = useQueryClient();
  
  return useMutation({
    mutationFn: (data: any) => api.createTutor(data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['tutors'] });
    },
  });
};
```

### Authentication Integration

#### Auth Provider

Create an authentication provider to manage user sessions:

```typescript
// src/contexts/AuthContext.tsx
import { createContext, useContext, useState, useEffect } from 'react';
import { frappe } from 'frappe-js-sdk';

interface AuthContextType {
  user: any;
  login: (email: string, password: string) => Promise<void>;
  logout: () => Promise<void>;
  isAuthenticated: boolean;
}

const AuthContext = createContext<AuthContextType | undefined>(undefined);

export const AuthProvider = ({ children }: { children: React.ReactNode }) => {
  const [user, setUser] = useState<any>(null);
  
  useEffect(() => {
    // Check for existing session
    const checkSession = async () => {
      try {
        const session = await frappe().auth.getSession();
        setUser(session.user);
      } catch (error) {
        setUser(null);
      }
    };
    checkSession();
  }, []);
  
  const login = async (email: string, password: string) => {
    const session = await frappe().auth.login({ usr: email, pwd: password });
    setUser(session.user);
  };
  
  const logout = async () => {
    await frappe().auth.logout();
    setUser(null);
  };
  
  return (
    <AuthContext.Provider value={{ user, login, logout, isAuthenticated: !!user }}>
      {children}
    </AuthContext.Provider>
  );
};

export const useAuth = () => {
  const context = useContext(AuthContext);
  if (!context) throw new Error('useAuth must be used within AuthProvider');
  return context;
};
```

```typescript
import { defineConfig } from 'vite';
import react from '@vitejs/plugin-react';
import path from 'path';

export default defineConfig({
  plugins: [react()],
  resolve: {
    alias: {
      '@': path.resolve(__dirname, './src'),
    },
  },
  server: {
    proxy: {
      '/api': {
        target: process.env.VITE_FRAPPE_URL || 'https://tuts.erpnext.zubbystudio.shop',
        changeOrigin: true,
        rewrite: (path) => path.replace(/^\/api/, ''),

    },
  },
});
```
